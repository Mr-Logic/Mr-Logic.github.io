hasOwnProperty()和in的区别

1.in操作符用来检查对象中是否存在某个属性(不区分实例属性和原型属性)

<script>

    //01 提供一个构造函数
    function Person(name) {
        this.name = name;
    }

    //02 设置构造函数的原型对象的属性
    Person.prototype.sayHello = function () {
        console.log("hello");
    }

    //03 创建对象
    var p1 = new Person();

    //04 使用in关键字判断对象中是否存在以下属性:name age sayHello
    console.log("age" in p1);       //false
    console.log("name" in p1);      //true
    console.log("sayHello" in p1);  //true

</script>

2.对象的hasOwnProperty方法用来检查对象中是否存在指定的属性(只检查实例属性)

<script>

    //01 提供一个构造函数
    function Person(name) {
        this.name = name;
    }

    //02 设置构造函数的原型对象的属性
    Person.prototype.sayHello = function () {
        console.log("hello");
    }

    Person.prototype.des = "默认的描述信息";

    //03 创建对象
    var p1 = new Person();

    //04 使用hasOwnProperty方法判断该属性是否是对象的实例属性
    console.log(p1.hasOwnProperty("age"));       //false
    console.log(p1.hasOwnProperty("name"));      //true
    console.log(p1.hasOwnProperty("sayHello"));  //false
    console.log(p1.hasOwnProperty("des"));       //false

</script>

3.判断某个属性是否存在且只存在于原型属性中

// 解决思路 :
// ① 属性存在于对象中
// ② 属性不是实例属性

<script>

    //01 提供一个构造函数
    function Person(name) {
        this.name = name;
    }

    //02 设置构造函数的原型对象的属性
    Person.prototype.sayHello = function () {
        console.log("hello");
    };

    Person.prototype.des = "默认的描述信息";

    //03 创建对象
    var p1 = new Person();


    function isProperty(obj, property) {
        return !obj.hasOwnProperty(property) && (property in obj);
    }

    console.log(isProperty(p1, "constructor"));    //true
    console.log(isProperty(p1, "age"));            //false
    console.log(isProperty(p1, "name"));           //false
    console.log(isProperty(p1, "des"));            //true
    console.log(isProperty(p1, "sayHello"));        //true
    console.log(p1);

</script>

react中使用echarts

1.安装echarts

yarn add echarts echarts-for-react --save
// echarts为核心类，echarts-for-react是为了组件化开发，避免new一个echarts类，多页面开发常需要new一个类
// 若Node版本过高可能会报错，此时需要降级
// node降级可以执行命令：npm install -g n，然后执行命令：n 版本号，即可将node降级为对应的版本
// 参考地址：https://www.npmjs.com/package/n

2.引入echarts

// 1.一种方式是全局引入
import echarts from 'echarts';    // 此种引入方式不好，会将echarts所有模块都引入
// 1.第二种方式是按需加载
import echarts from 'echarts/lib/echarts';    // 核心库，必需
import 'echarts/lib/chart/bar';                // bar图
import 'echarts/lib/component/tooltip';        // 加载需要的组件
import 'echarts/lib/component/title';
import 'echarts/lib/component/legend';
import 'echarts/lib/component/markPoint';
// 2.如果需要更改主题，还需要引入主题，如不需要忽略此步使用默认主题(主题可在官网上直接下载或配置后下载)
import themes from '主题文件所在目录';
// 3.引入echarts-for-react    // 必需
import ReactEcharts from 'echarts-for-react';

3.如需要更改主题，则注入主题

componentsWillMount () {
    echarts.registerTheme('xxx', themes);
}

4.使用，以柱形图为例，参考官方文档即可

// render函数中使用
<ReactCharts 
    theme="xxx"
    options={this.getOptions()}
    style={{height: '500px'}}
/>

// 定义getOptions方法
getOptions = () => {
    let option = {
        title: {
            text: '标题名称'
        },
        // 副标题
        legend: {
            data: ['A订单量', 'B订单量']
        },
        // x轴展示的数据
        xAxis: {
            data: ['周一', '周二', '周三', '周四', '周五']    // 将data替换为服务端返回的数据
        }，
        // y轴
        yAxis: {
            type: 'value'
        },
        series: [
            {
                name: 'A订单量',
                type: 'bar',
                data: [1000, 2000, 3000, 1500, 2000]    // 需要和x轴的data保持个数一致
            },
            //series中一个对象代表一组数据，如需要同时显示多组数据对比情况，可定义多个对象
            {
                name: 'B订单量',
                type: 'bar',
                data: [100, 200, 500, 100, 2000]    // 需要和x轴的data保持个数一致
            }
        ],
        // 鼠标移入每项数据后显示的modal
        tooltip: {
            trigger: 'axis'
        }
    };

    return option;
}

4.使用，以饼图为例

          importLoaders: 1,

                '>1%',
                'last 4 versions',
                'Firefox ESR',
                'not ie < 9', // React doesn't support IE8 anyway
              ],
              flexbox: 'no-2009',
            }),
          ],
        },
      }
    ],
  },
}

将上面的代码修改两处地方：最开始的css改为less，最后添加{loader: require.resolve(‘less-loader’)}

4.安装antd

yarn add antd

5.安装babel-plugin-import实现antd按需加载

yarn add babel-plugin-import

6.修改webpack配置以支持antd按需加载

####在webpack.config.dev.js中找到如下代码
{
test: /.(js|jsx|mjs)$/,
include: paths.appSrc,

        compact: true,

在上面的代码中的options中添加配置，修改后为如下即可：

    },
}

create-react-app脚手架创建的项目开发过程知识点总结

1.create-react-app创建的项目，发送ajax请求时，若请求路径找不到，脚手架开启的node服务会自动在public文件夹下去寻找。因此，在后端接口完成之前可在此文件夹下创建模拟数据供前端使用，项目上线时将模拟文件删除即可。

2.使用react路由

yarn add react-router-dom

3.react的css是全局的，为避免相互影响，可以使用styled-components将css样式转换为js样式组件

yarn add styled-components

4.react中使用antd框架的DatePicker组件时，获取的是moment时间对象，需要安装moment，然后转为字符串

yarn add moment
import moment from 'moment';
const sDateS = moment(sDate)    // sDate为获取的moment时间对象, moment()即可将其转为字符串

5.less中设置宽高需要计算100%和px或rem和差时避免100和px前的数值直接计算得到错误结果

width: (100% - 1.2rem);    //计算结果为width(98.8%)
// 需要改为以下计算方式：
width: (~'100% - 1.2rem');
height: (~'100% - 1.2px');

在redux中使用redux-thunk中间件来处理异步请求

使用redux-thunk中间件来管理action和store之间的dispatch，在创建store的时候使用

yarn add redux-thunk

import { createStore, compose, applyMiddleware } from 'redux';
import reducer from './reducer';
import thunk from 'redux-thunk';

const composeEnhancers = window.__REDUX_DEVTOOLS_EXTENSION_COMPOSE__ || compose;
const store = createStore(reducer, /* preloadedState, */ composeEnhancers(
    applyMiddleware(thunk)
));

export default store;

然后即可在组件的action中派发请求

const mapDispatchToProps = (dispatch) => {
    return {
        handleInputFocus () {
            dispatch(actionCreators.getList());
        }
    }
}

在组件的actionCreators文件中发送异步请求

export const changeList = (data) => ({
    type: 'change_list',
    data
})

export const getList = () => {
    // 使用redux-thunk后此处即可返回一个函数
    return (dispatch) => {
        axios.get('/api/headerList.json').then((res) => {
            const data = res.data;
            dispatch(changeList(data.data))
        }).catch((err) => {
            console.log(err)
        })
    }
}

react项目中使用react-redux基本步骤

1.引入redux

yarn add redux

2.引入react-redux

yarn add react-redux

3.创建store文件夹，store中创建index.js、reducer.js，完成store基本结构

index.js文件中

import { createStore } from 'redux';
import reducer from './reducer';

const store = createStore(reducer);

export default store;

reducer.js文件中

const defaultState = {
    // 此处存储的是初始化的store数据，即原来存在组件state中的数据
    focused: false    //假设现在有一个focused的数据
};

export default (state=defaultState, action) => {
    if (action.type === 'search_focus') {
        return {
            focused: true
        }
    }
    if (action.type === 'search_blur') {
        return {
            focused: false
        }
    }
    return state;
}

4.根组件App.js中引入store和Provider并包裹子组件，则任意子组件都可使用store

import { Provider } from 'react-redux';
import store from './store'
class App extends Component {
      render() {
        return (
          <Provider store={store}>
            <内部组件 />
          </Provider>
        );
      }
}

5.在子组件中使用connect连接组件和store

import { connect } from 'react-redux';

// 原来组件中使用this.state.focused的地方则可替换为this.props.focused
// 原来组件中调用的方法this.handleInputFocus的地方则可替换为this.props.handleInputFocus

const mapStateTopProps = (state) => {
    return {
        focused: state.focused    // 将store中的focused映射到组件的props中去
    }
}
const mapDispatchToProps = (dispatch) => {
    return {
         handleInputFocus () {
            // 创建action
            const action = {
                type: 'search_focus'
            };
            // 将action派发给store
            dispatch(action);
        },
        handleInputBlur () {
            const action = {
                type: 'search_blur'
            };
            dispatch(action);
        }
    }
}
export default connect(mapStateToProps, mapDispatchToProps)(子组件);

6.当需要管理的数据较多时，可将store中的reducer拆分到各个组件中，再用combineReducers连接起来。

每个组件的store文件夹中都可以有入口store文件index.js、处理文件reducer.js、常量文件constants.js、action创建文件actionCreators.js，将其他三个文件导入到index.js文件中再导出，外部需要使用时直接访问index.js文件即可。

import { combineReducers } from 'redux';
import headerReducer from '../common/header/store/reducer';

export default combineReducers({
    header: headerReducer
})

需要注意的是组件中的数据多了一层结构

const mapStateTopProps = (state) => {
    return {
        focused: state.header.focused    // 将store中的focused映射到组件的props中去
    }
}

7.由于reducer中不可直接更改state,可使用immutable插件来讲state包装为immutable对象

yarn add immutable
yarn add redux-immutable

在组件的reducer.js中

import { fromJS } from 'immutable';
// 原来的defaultState改为
const defaultState = fromJS({
    focused: false
});

export default (state=defaultState, action) => {
    if (action.type === constants.SEARCH_FOCUS) {
        // immutable对象的set方法，会结合之前的immutable对象的值和设置的值，返回一个全新的对象
        return state.set('focused', true)    // 此时更改state需要使用immutable对象的set方法
    }
    if (action.type === constants.SEARCH_BLUR) {
        return state.set('focused', false)
    }
    return state;
}

在组件中获取state也需要immutable方法

// 由于state是redux-immutable中的combineReducers函数创建的对象，故为immutable对象，需要使用get方法
const mapStateToProps = (state) => {
    return {
        focused: state.getIn(['header', 'focused'])    // 等价于state.get('header').get('focused)
    }
}
